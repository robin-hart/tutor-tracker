% JEDES MAL ändern (Stunden * 60min/h):
\pgfmathtruncatemacro{\uebertragVormonat}{150}

% Counter Zeitübertrag aus dem Vormonat 
\newcounter{variablezeituebertragvormonat}
\setcounter{variablezeituebertragvormonat}{\uebertragVormonat}

\newcommand{\zeituebertragvormonat}{
    \pgfmathtruncatemacro{\hours}{\value{variablezeituebertragvormonat}/60}
    \pgfmathtruncatemacro{\minutes}{\value{variablezeituebertragvormonat}-\hours*60}
    \hours h \ifnum\minutes<10 0\fi\minutes min
}


% Counter für IST-Arbeitszeit in Minuten
\newcounter{totalminutes}
\setcounter{totalminutes}{0}


% Befehl zum Hinzufügen von Zeit (Stunden, Minuten)
\newcommand{\addtime}[2]{%
    % Gesamtminuten berechnen
    \pgfmathtruncatemacro{\tmpminutes}{#1*60+#2}%
    \addtocounter{totalminutes}{\tmpminutes}%
    % Zeit in der Tabelle ausgeben
    #1h \ifnum#2<10 0\fi#2min%
}

% Befehl um Gesamtsumme auszugeben
\newcommand{\istarbeitszeit}{
    \pgfmathtruncatemacro{\hours}{\value{totalminutes}/60}
    \pgfmathtruncatemacro{\minutes}{\value{totalminutes}-\hours*60}
    \hours h \ifnum\minutes<10 0\fi\minutes min
}


% Counter für Summe (IST+Übertrag)
\newcommand{\summeistunduebertrag}{
    \pgfmathtruncatemacro{\hours}{(\value{totalminutes} + \value{variablezeituebertragvormonat})/60}
    \pgfmathtruncatemacro{\minutes}{(\value{totalminutes} + \value{variablezeituebertragvormonat})-\hours*60}
    \hours h \ifnum\minutes<10 0\fi\minutes min
}

% Constant für SOLL-Arbeitszeit in Minuten
\pgfmathtruncatemacro{\sollarbeitszeit}{1560} % 26h * 60min/h = 1560min

% Counter für SOLL-Arbeitszeit in Minuten
\newcounter{variablesollarbeitszeit}
\setcounter{variablesollarbeitszeit}{\sollarbeitszeit}


% Counter für Zeitübertrag auf nächsten Monat
\newcommand{\zeituebertragnaechstermonat}{
    \pgfmathtruncatemacro{\hours}{(\value{totalminutes} + \value{variablezeituebertragvormonat} - \value{variablesollarbeitszeit})/60}
    \pgfmathtruncatemacro{\minutes}{abs((\value{totalminutes} + \value{variablezeituebertragvormonat} - \value{variablesollarbeitszeit})-\hours*60)}
    \hours h \ifnum\minutes<10 0\fi\minutes min
}
